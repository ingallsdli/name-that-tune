      <input id="guessInput" placeholder="Guess the song title..." />

      <div id="feedback"></div>

      <div class="buttons">
        <button id="submitBtn">Submit</button>
        <button id="revealBtn">Reveal</button>
        <button id="nextBtn">Next</button>
      </div>
    </div>

    <script>
      const audio = document.getElementById("audio");
      const playBtn = document.getElementById("playBtn");
      const submitBtn = document.getElementById("submitBtn");
      const revealBtn = document.getElementById("revealBtn");
      const nextBtn = document.getElementById("nextBtn");
      const guessInput = document.getElementById("guessInput");
      const feedback = document.getElementById("feedback");
      const roundEl = document.getElementById("round");
      const scoreEl = document.getElementById("score");

      let songs = [];
      let currentSong = null;
      let round = 1;
      let score = 0;

      function normalize(str) {
        return str.toLowerCase().replace(/[^a-z0-9 ]/g, "").trim();
      }

      async function loadSongs() {
        const res = await fetch("https://itunes.apple.com/search?term=disney&entity=song&limit=25");
        const data = await res.json();

        songs = data.results
          .filter(s => s.previewUrl)
          .map(s => ({
            title: s.trackName,
            artist: s.artistName,
            audioClip: s.previewUrl
          }));

        pickSong();
      }

      function pickSong() {
        if (!songs.length) return;

        currentSong = songs[Math.floor(Math.random() * songs.length)];
        audio.src = currentSong.audioClip;
        audio.load();

        feedback.textContent = "";
        guessInput.value = "";
      }

      playBtn.onclick = () => {
        audio.currentTime = 0;
        audio.play().catch(err => console.error(err));
      };

      submitBtn.onclick = () => {
        const guess = normalize(guessInput.value);
        const correct = normalize(currentSong.title);

        if (guess && (guess === correct || correct.includes(guess))) {
          feedback.textContent = "✅ Correct!";
          score++;
          scoreEl.textContent = score;
        } else {
          feedback.textContent = "❌ Incorrect!";
        }
      };

      revealBtn.onclick = () => {
        feedback.textContent = `Answer: ${currentSong.title} by ${currentSong.artist}`;
      };

      nextBtn.onclick = () => {
        round++;
        roundEl.textContent = round;
        pickSong();
      };

      loadSongs();
    </script>
